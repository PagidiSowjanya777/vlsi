// Code your design here
module d_flipflop (
    input wire clk,      // Clock signal
    input wire rst_n,    // Active-low asynchronous reset
    input wire d,        // Data input
    output reg q         // Output
);

    // Triggered on the rising edge of the clock OR falling edge of reset
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            q <= 1'b0;   // Reset the output to 0 immediately
        end else begin
            q <= d;      // Transfer input D to output Q on clock edge
        end
    end
endmodule

`timescale 1ns / 1ps  // Set the time unit to 1ns and precision to 1ps

module tb_d_flipflop;

    // Inputs (declared as reg because we drive values into them)
    reg clk;
    reg rst_n;
    reg d;

    // Outputs (declared as wire to observe the design's output)
    wire q;

    // Instantiate the Unit Under Test (UUT)
    d_flipflop uut (
        .clk(clk),
        .rst_n(rst_n),
        .d(d),
        .q(q)
    );

    // Clock generation: Toggle clock every 5ns (creates a 10ns period / 100MHz clock)
    always begin
        #5 clk = ~clk;
    end

    // Stimulus block
    initial begin
        // Initialize inputs
        clk = 0;
        rst_n = 0; // Start in reset state
        d = 0;

        // Hold reset state for 20ns
        #20;
        rst_n = 1; // Release reset
        
        // Apply test cases
        #10 d = 1; // Set D to 1
        #10 d = 0; // Change D to 0
        #15 d = 1; // Change D between clock edges
        
        // Test asynchronous reset while D is high
        #10 rst_n = 0; 
        #10 rst_n = 1; // Release reset again
        
        #20;
        $finish; // End the simulation
    end

    // Optional: Monitor changes in the console
    initial begin
      $dumpfile("dump.vcd"); // Name of the file where data will be stored
        $dumpvars(0, tb_d_flipflop);
        $monitor("Time = %0t | rst_n = %b | d = %b | q = %b", $time, rst_n, d, q);
    end

endmodule
