// 1. The Buggy Code with a Race Condition
module race_condition (
    input wire clk,
    input wire d,
    output reg a,
    output reg b
);

    // Block 1: Updates 'a' on clock edge using BLOCKING assignment
    always @(posedge clk) begin
        a = d;
    end

    // Block 2: Reads 'a' on the same clock edge using BLOCKING assignment
    always @(posedge clk) begin
        b = a; // RACE CONDITION
    end

endmodule

// 2. The Fixed Code (Race-Free)
module race_free (
    input wire clk,
    input wire d,
    output reg a,
    output reg b
);

    // Block 1: Uses non-blocking assignment
    always @(posedge clk) begin
        a <= d;
    end

    // Block 2: Correctly samples the value of 'a' before the clock edge
    always @(posedge clk) begin
        b <= a; // Deterministic
    end

endmodule// Code your testbench here
