module pulse_generator (
    input  wire clk,
    input  wire rst,
    input  wire signal_in,
    output wire pulse_out
);

reg signal_d;

// Store previous value of signal_in
always @(posedge clk or posedge rst)
begin
    if (rst)
        signal_d <= 1'b0;
    else
        signal_d <= signal_in;
end

// Generate one-clock pulse on rising edge
assign pulse_out = signal_in & ~signal_d;

endmodule

`timescale 1ns/1ps

module pulse_generator_tb;

reg clk;
reg rst;
reg signal_in;
wire pulse_out;

// Instantiate DUT
pulse_generator uut (
    .clk(clk),
    .rst(rst),
    .signal_in(signal_in),
    .pulse_out(pulse_out)
);

// Clock Generation (10 ns period)
always #5 clk = ~clk;

initial
begin
    clk = 0;
    rst = 1;
    signal_in = 0;

    // Release reset
    #12;
    rst = 0;

    // Rising edge -> Pulse
    #8;
    signal_in = 1;

    // Hold High
    #20;
    signal_in = 1;

    // Falling edge
    #10;
    signal_in = 0;

    // Rising edge again -> Pulse
    #20;
    signal_in = 1;

    // Falling edge
    #10;
    signal_in = 0;

    // Rising edge again -> Pulse
    #20;
    signal_in = 1;

    #20;
    $finish;
end

// Display values
initial begin
  $dumpfile("dump.vcd");
  $dumpvars(0, pulse_generator_tb);
  
    $display("Time\tclk\trst\tsignal_in\tpulse_out");
    $monitor("%0t\t%b\t%b\t%b\t\t%b",
             $time, clk, rst, signal_in, pulse_out);
end

endmodule
